CREATE DATABASE "nksdb"
CREATE TABLE `nksdb`.`medialist` ( `listid` INT(11) NOT NULL , `medialist` VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , PRIMARY KEY (`listid`)) ENGINE = InnoDB;
CREATE TABLE `nksdb`.`userdata` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `userid` VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `password` VARCHAR(100) NOT NULL , `name` VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `address` VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
CREATE TABLE `nksdb`.`media` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `title` VARCHAR(100) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `fname` TEXT CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `extension` TEXT CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `raw_data` LONGBLOB NOT NULL , `text` TEXT CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `listid` INT(11) NOT NULL , `userid` INT(11) NOT NULL , PRIMARY KEY (`id`) , FOREIGN KEY (`listid`) REFERENCES `medialist`(`listid`) ON DELETE RESTRICT ON UPDATE CASCADE, FOREIGN KEY (`userid`) REFERENCES `userdata`(`id`) ON DELETE RESTRICT ON UPDATE CASCADE) ENGINE = InnoDB;
